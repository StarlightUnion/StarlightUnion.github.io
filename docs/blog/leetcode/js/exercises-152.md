var maxProduct = function(nums) {
  let n;

  for (let i = 0; i < nums.length; i++) {
    let _n = nums[i];
    if (n !== undefined) {
      n = _n > n ? _n : n;
    } else {
      n = _n;

    for (let j = i + 1; j < nums.length; j++) {

      n = _n > n ? _n : n;
    }
  }

  return n;
};
