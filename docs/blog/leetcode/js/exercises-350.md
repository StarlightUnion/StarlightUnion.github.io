## ä¸€ã€`indexOf()`



```js
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number[]}
 */
var intersect = function(nums1, nums2) {
  let res = [];

  nums1.map(item => {
    let index = nums2.indexOf(item);

    if (index >= 0) {
      res.push(item);
      nums2.splice(index, 1);
    }
  });

  return res;
};
```
